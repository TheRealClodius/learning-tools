# MemoryOS MCP Server Environment Variables Sample
# Copy this to your .env.local file and fill in your actual API keys

# Existing API Keys (already configured)
ANTHROPIC_API_KEY=your_anthropic_api_key_here
PERPLEXITY_API_KEY=your_perplexity_api_key_here  
OPENWEATHER_API_KEY=your_openweather_api_key_here

# NEW: Required for Run Summary Generation
GEMINI_API_KEY=your_gemini_api_key_here

# MemoryOS MCP Server Configuration (SSE-based)
MEMORYOS_SERVER_URL=https://your-memoryos-app.railway.app/sse
MEMORYOS_TIMEOUT=30

# MemoryOS User Configuration
MEMORY_USER_ID=default_user

# Notes:
# - MEMORYOS_SERVER_URL: URL of your MemoryOS SSE-based MCP server (Railway deployment)
# - MEMORYOS_TIMEOUT: Request timeout in seconds (default: 30)
# - MEMORY_USER_ID: Unique identifier for your user in the memory system
# - GEMINI_API_KEY: Required for intelligent run summaries (tool usage, reasoning)
#
# To deploy your MemoryOS server:
# 1. Clone: https://github.com/TheRealClodius/Memory-Signal
# 2. Deploy to Railway (no API key needed for SSE-based MCP server)
# 3. Update MEMORYOS_SERVER_URL above with your Railway app URL + /sse
# 4. Get Gemini API key from: https://aistudio.google.com/app/apikey
#
# Enhanced Memory Features (SSE-based MCP):
# - Automatic Q&A storage with intelligent run summaries
# - Run summaries include: reasoning approach, tools used, what worked/failed
# - Generated by Gemini Flash 2.5 for optimal performance
# - Fallback summary when Gemini unavailable
# - SSE-based MCP protocol for real-time communication
#
# Performance benefits:
# - Real-time SSE connection with automatic reconnection
# - Multi-user isolation and scalability
